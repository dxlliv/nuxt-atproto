The module stores OAuth sessions in the browser via `@atproto/oauth-client-browser`. You can keep multiple accounts and switch between them with `restore(did)` — no full OAuth redirect required for accounts already authorized.

Reference: [`useStoredProfiles`](https://github.com/dxlliv/nuxt-atproto/blob/main/playground/composables/useStoredProfiles.ts), [`usePlaygroundSession`](https://github.com/dxlliv/nuxt-atproto/blob/main/playground/composables/usePlaygroundSession.ts), [`PlaygroundDemo.vue`](https://github.com/dxlliv/nuxt-atproto/blob/main/playground/components/PlaygroundDemo.vue).

## `restore()` vs `signIn()`

| Method | Use when |
|--------|----------|
| `signIn()` / `signInWithHandle()` | New account or re-authorization (OAuth redirect) |
| `restore(did)` | Switch to a session the OAuth client already stored for that DID |

```ts
const { restore } = useAtprotoAuth()

const session = await restore('did:plc:your-did-here')
// session.sub === did; authenticated agent cache is invalidated
```

`restore()` emits `atproto:sessionRestored` immediately (not deferred to `app:mounted`).

## Cache profile metadata locally

OAuth sessions expose the DID (`session.sub`) but not display name or avatar. Most apps cache profile fields in `localStorage` (or IndexedDB) keyed by DID:

```ts [composables/useAccountProfiles.ts]
const STORAGE_KEY = 'my-app:profiles'

interface CachedProfile {
  did: string
  handle: string
  displayName?: string
  avatar?: string
}

export function useAccountProfiles() {
  const profiles = useState<Record<string, CachedProfile>>('account-profiles', () => ({}))

  function loadFromStorage() {
    if (!import.meta.client) return
    try {
      profiles.value = JSON.parse(localStorage.getItem(STORAGE_KEY) || '{}')
    }
    catch {
      profiles.value = {}
    }
  }

  function upsert(profile: CachedProfile) {
    profiles.value = { ...profiles.value, [profile.did]: profile }
    localStorage.setItem(STORAGE_KEY, JSON.stringify(profiles.value))
  }

  function remove(did: string) {
    const next = { ...profiles.value }
    delete next[did]
    profiles.value = next
    localStorage.setItem(STORAGE_KEY, JSON.stringify(profiles.value))
  }

  return { profiles, loadFromStorage, upsert, remove }
}
```

## Hook profile fetch to session lifecycle

After login or restore, fetch and cache the profile; on delete, remove the cache entry:

```ts
const { upsert, remove } = useAccountProfiles()

async function fetchProfile(did: string) {
  const agent = useAtprotoAgent('public')
  const { data } = await agent.getProfile({ actor: did })
  upsert({
    did: data.did,
    handle: data.handle,
    displayName: data.displayName,
    avatar: data.avatar,
  })
}

hook('atproto:sessionCreated', fetchProfile)
hook('atproto:sessionRestored', fetchProfile)
hook('atproto:sessionDeleted', remove)
```

## Account switcher UI

List cached profiles excluding the active DID, and call `restore()` on switch:

```vue
<script setup lang="ts">
const { session } = useAtprotoSession()
const { restore } = useAtprotoAuth()
const { profiles, loadFromStorage } = useAccountProfiles()

onMounted(loadFromStorage)

const otherAccounts = computed(() =>
  Object.values(profiles.value).filter(p => p.did !== session.value?.sub),
)
</script>

<template>
  <ul>
    <li v-for="account in otherAccounts" :key="account.did">
      {{ account.displayName ?? account.handle }}
      <button type="button" @click="restore(account.did)">
        Switch
      </button>
    </li>
  </ul>
</template>
```

## Add another account

Keep the logged-in UI visible and offer a second sign-in flow (`signIn()` or `signInWithHandle()`). The OAuth client stores each authorized DID; after redirect, the new session becomes active. Cached profiles for previous accounts remain until you remove them.

## Cleanup on sign-out

`signOut()` revokes the **current** session only. Other stored OAuth sessions remain available via `restore()`. Use `atproto:sessionDeleted` to drop cached profile rows when the OAuth client removes a session (revoke, expiry, etc.).
